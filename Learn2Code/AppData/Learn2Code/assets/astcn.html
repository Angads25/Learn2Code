Program to create a class complex with member function to add and subtract two complex numbers.

class complex
{   int r,i;
     
     public:
     void getdata(){cout<<"Enter the vaue of real and imaginary part ";cin>>r>>i;}

     complex sum(complex c)
     {    complex t;
           t.r=r+c.r;
           t.i=i+c.i;
           return(t);
     }
     complex sub(complex c)
     {    complex t;
           t.r=r-c.r;
           t.i=i-c.i;
           return(t);
     }
};
