Program to convert Farenheit temperature to Centigrate or Centigrate to Fahrenheit.

#include<stdio.h>
#include<conio.h>
void main()
{  char ch;
   float fh,cl;
   Start:
   clrscr();
   printf("1.Convert Fahrenheit to Centigrate");
   printf("\n2.Convert Centigrate to Fahrenheit");
   printf("\nChoose an option:");
   ch=getchar();
   switch(ch)
   {  case '1':clrscr();
           printf("Enter the Fahrenheit value:");
           scanf("%f",&fh);
           cl=(fh-32)/1.8;
           printf("Corresponding Celsius value is:%f",cl);
           break;

      case '2':clrscr();
           printf("Enter the Celsius value:");
           scanf("%f",&cl);
           fh=(1.8*cl)+32;
           printf("Corresponding Fahrenheit value is:%f",fh);
           break;

      default :printf("Please enter a valid option");
           goto Start;
   }
   getch();
}
