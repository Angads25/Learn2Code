Program to find the sum of the following series upto n terms:
1+1/2+1/3+1/4 ..........1/n.

#include<iostream.h>
#include<stdio.h>
void main()
{   float i,n,sum=0;
     clrscr();
     cout<<"Enter the value of n";
     cin>>n;
     i=1;
     while (i<=n)
     {    sum=sum+1/i;
           i++;
     }
     cout<<"The sum of this series for give number of terms is "<<sum;
     getch();

}
