Program to create a class time to show and take time in hours and minutes. Also create member functions to add two times and show the result.

{   int min,hr;
     
     public:

    void sumtime(time t1,time t2)

         a=t1.hr+t2.hr; 
         b=t1.min+t2.min;
         if(b>60)
         {   hr=a+b/60;
              min=b-(b/60)*60;
         }
         if(hr>=24)
         hr=hr%24;
    }
    void showtime()
    {   cout<<"\nThe time is "<<hr<<": "<<min;
    }
};
