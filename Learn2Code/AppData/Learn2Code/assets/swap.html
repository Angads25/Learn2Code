Program to swap the values of two numbers by value, by pointers and by alias.

#include<conio.h>
#include<iostream.h>
void swap(int a,int b)
{   int y=a;
     a=b;
     b=y;
     cout<<"The values after swaping by value are "<<a<<" & "<<b;
}
void swap(int *j,int *k)
{   int y;
     y=*j;
     *j=*k;
     *k=y;
     cout<<"The values after swaping by address are "<<*j<<" & "<<*k;
}
void swap2(int & j,int & k)
{   int z=j;
     j=k;
     k=z;
     cout<<"The values after swaping by alias are " <<j<<" & "<<k;
}
void main()
{   clrscr();
     int x,y,a,b,c,d;
     cout<<"\nEnter the values of x and y to swap by value ";
     cin>>x>>y;
     swap(x,y);
     cout<<"\nOriginal values after swaping x and y by value is "<<x<<" & "<<y;
     cout<<"\nEnter the values of a and b to swap by address ";
     cin>>a>>b;
     swap(&a,&b);
     cout<<"\nOriginal values after swaping a and b by address is "<<a<<" & "<<b;
     cout<<"\nEnter the values of c and d to swap by alias ";
     cin>>c>>d;
     swap2(c,d);
     cout<<"\nOriginal values after swaping c and d by alias is "<<c<<" & "<<d;
     getch();
}
