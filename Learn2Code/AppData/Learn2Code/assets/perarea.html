Program to find perimeter of triangle, rectangle, square and circle by function overloading.

#include<conio.h>
#include<iostream.h>
void peri(float a)
{   float c=4*a;
     cout<<"Perimeter of square is "<<c;
}
void peri(float l,float b)
{   float c=2*(l+b); 
     cout<<"Perimeter of the rectangle is "<<c;
}
void peri(float s1,float s2,float s3)
{   float s=s1+s2+s3;
     cout<<"Perimeter of the triangle is "<<s;
}
void peri()
{   float c=(2*22*14)/7;
     cout<<"\nPerimeter of the circle is "<<c;
}
