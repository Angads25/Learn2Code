Program to accept a three digit number & find the sum of the digits of the number without using loop.

 #include<stdio.h>
#include<conio.h>
void main()
{  int n,h,t,o,sum;
   clrscr();
   printf("Enter a 3 digit no.:");
   scanf("%d",&n);
   h=n/100;
   t=(n-(h*100))/10;
   o=n%10;
   sum=h+t+o;
   printf("Sum of all the digits is:%d",sum);

}
